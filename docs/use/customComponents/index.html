customComponents 自定义组件

自定义组件示例​

如下，首先我们先申明两个组件CustomField1和CustomField2，其次可以申明一个对象customComponents，该对象中的属性名custom1、custom2与unitedSchema中定义的唯一组件 key 值对应，否则表单无法渲染自定义组件。

import React, { useState, memo } from 'react'
import { render } from 'react-dom'
import DripForm from '@jdfed/drip-form'
import dripTheme from '@jdfed/drip-form-theme-antd'
import unitedSchema from './unitedComponents.json'
import '@jdfed/drip-form-theme-antd/dist/index.css'

const CustomField1 = memo(() => {
  const [count, setCount] = useState(0)
  return (
    <div>
      <p>You clicked {count} times</p>
      <div onClick={() => setCount(count + 1)}>Click me</div>
    </div>
  )
})

const CustomField2 = memo(() => {
  return <div>这是自定义组件2</div>
})

const customComponents = {
  custom1: CustomField1,
  custom2: CustomField2,
}

const Form = () => {
  return (
    <div>
      <DripForm
        unitedSchema={unitedSchema}
        uiComponents={{
          'drip-theme': dripTheme,
        }}
        customComponents={customComponents}
      />
    </div>
  )
}

配置 unitedSchema​

同时，我们需要在unitedSchema中配置一下，只需要将Schema中ui的type属性设置成我们在customComponents中配置的属性key值，在本示例中即为custom1和custom2。

unitedSchema.json
{
  "validateTime": "submit",
  "type": "object",
  "theme": "drip-theme",
  "schema": [
    {
      "title": "",
      "type": "string",
      "ui": {
        "type": "custom1"
      },
      "fieldKey": "custom_click1"
    },
    {
      "type": "string",
      "title": "",
      "ui": {
        "type": "custom2"
      },
      "fieldKey": "custom_component2"
    }
  ]
}
